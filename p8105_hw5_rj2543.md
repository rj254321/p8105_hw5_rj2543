file_read = function(x){
    path = str_c("./data/", x)
    
    y = read_csv(path) %>% 
      mutate(patient = x) %>%
      separate(patient, into = c("id", "remove"), sep = ".c") %>% 
      select(-remove) %>% 
      select(id, everything())
    y
}

patient = file %>% 
  mutate(info = map(file$names, file_read)) %>% 
  separate(names, into = c("patient_id", "remove1"), sep = ".c") %>%

Make a spaghetti plot showing observations on each subject over time, and comment on differences between groups.

``` r
plot_patient = function(a){
  patient_tidy %>%
    gather(key = week, value = obs, week_1:week_8) %>%
    separate(week, into = c("remove", "week"), sep = "_") %>% 
    select(-remove) %>% 
    filter(id == a) %>% 
    ggplot(aes(x = week, y = obs, group = arm, color = arm)) +
    geom_point() +
    geom_path() +
    labs(
      title = "Observations over time among control and experimental arms",
      x = "Week",
      y = "Observations",
      caption = str_c("Patient ", a)
    )
}

b = 1:10

map(patient_tidy %>% distinct(id) %>% pull(), plot_patient)
```
